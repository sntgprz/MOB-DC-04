// Constant Declaration (inferred type). Remember IMMUTABLE.
let name = "Tedi"

// Variable Declaration (inferred type). Remember MUTABLE. 
var age = 21

// Type Annotation (string)
var country: String = "USA"

// Integers are whole numbers (no decimal)
let questionOfLife: Int = 42

// Floats and doubles: numbers with fractional components. Double has a precision of 15 decimal digits whereas double has a precision of 6. Double is inferred by default.
let decimalNum: Double = 55.23

// Booleans are true false values
let truth = true
var lie = false

// Print something out to console
print("Hello there")
// Print without the ending new line character
print("hello", terminator: "")

// String Interpolation, also known as string literal
print("Hello \(name)")

// Conditional statements
var age = 59
if age >= 55 {
  print("You can drink, but don't hurt yourself!")
} else if age >= 21 && age < 55 {
  print("You can drink but not under senior discount")
} else {
  print("You can't drink at all.")
}

// While loops
var num = 10
while num < 20 {
  print("Counting at \(num)"
  num++
}

// For loop inclusive prints 1 to 10
for i in 1..10 {
  print(i)
}

// For loop with half open range operator. Prints 1 to 9
for i in 1..<10 {
  print(i)
}
